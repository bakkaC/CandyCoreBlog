下面给你 **深拷贝（deepClone）** 和 **对象扁平化（flattenObject）** 的实现示例。  
两者都包含核心复杂点：

- ✔ 循环引用处理
- ✔ 特殊对象（Date、RegExp）的基础处理
- ✔ 对象 / 数组区分

---


```js
function isObject(val) {
  return val !== null && typeof val === 'object';
}

function isSpecial(val) {
  return val instanceof Date || val instanceof RegExp || val instanceof Array;
}

function cloneSpecial(val) {
  if (val instanceof Date) return new Date(val.getTime());
  if (val instanceof RegExp) return new RegExp(val);
  if (val instanceof Array) return new Array(val);
}
```

深拷贝结束条件：
- 基本类型
- 函数
- 循环引用

特殊处理：instanceof
- Date/RegExp/Array，直接 new 一个新的
- Map/Set

```js
function deepClone(obj, seen = new WeakMap()) {
  // 1. 基本类型直接返回
  if (!isObject(obj)) return obj;

  // 2. 循环引用检查
  if (seen.has(obj)) return seen.get(obj);

  // 3. 特殊对象（Date / RegExp）
  if (isSpecial(obj)) return cloneSpecial(obj);

  // 4. 初始化容器 {} 或 []
  const result = Array.isArray(obj) ? [] : {};
  seen.set(obj, result);

  // 5. 遍历子节点
  for (const key in obj) {
    if (obj.has0wnProperty(key)) {
        result[key] = deepClone(obj[key], seen);
    }
  }

  return result;
}
```

对象扁平化：
→ `{ a: { b: 1 } }` → `{ 'a.b': 1 }`  
→ 嵌套对象转成扁平键值对

Q2: 怎么拼接 key？  
→ 递归时传递 prevKey  
→ 新 key = `${prevKey}.${currentKey}`

```js
function flatten(obj, depth = Infinity) {
  const res = {}
  const seen = new WeakMap()

  const dfs = (cur, prekey, d) => {
    // 1. 基本类型，或者到达目标深度
    if (!isObject(cur) || d <= 0) {
      res[prekey] = cur
      return
    }
	// 2. 循环引用检查
    if (seen.has(cur)) {
      res[prekey] = '[Circular]'
      return
    }
    seen.set(cur, true)
	// 3. 特殊对象（Date, RegExp），在这个算法中和普通对象的处理一样，直接简化掉了
	
	// 4. 对象或数组：继续深入
    for (const key in obj) {
	  if (obj.hasOwnProperty(key)) {
	    const newPath = prekey ? `${prekey}.${key}` : key
	    dfs(obj[key], newPath, d - 1)
	  }
    }
  
  dfs(obj, '', depth)
  return res
}
```

---

为什么深拷贝需要重新创建特殊对象，因为他的目的是创建新对象而不是修改原对象
如果当作普通对象，直接赋值，会造成两个对象共享同一个空间的问题

而对象扁平化是直接修改或者覆盖原对象，不用考虑原对象和新对象值共享的问题

数组扁平化：常规的递归考察

```js
function flatten(arr) {
  const res = [];

  for (const item of arr) {
    if (Array.isArray(item)) {
      res.push(...flatten(item));
    } else {
      res.push(item);
    }
  }

  return res;
}

// reduce 写法
function flatten(arr) {
  return arr.reduce((acc, cur) => {
    return acc.concat(
      Array.isArray(cur) ? flatten(cur) : cur
    );
  }, []);
}

// 按深度展开
function flatten(arr, depth = 1) {
  if (depth <= 0) return [...arr]
  
  const res = []
  for (const item of arr) {
    if (Array.isArray(item)) {
      res.push(...flatten(item, depth - 1))
    } else {
      res.push(item)
    }
  }
  return res
}

function flattenReduce(arr, depth = 1) {
  return depth > 0
    ? arr.reduce((acc, item) => 
	acc.concat(Array.isArray(item) ? flattenReduce(item, depth - 1) : item), [])
    : [...arr]
}
```

``` js
// 最简写法
function unique(arr) {
  return [...new Set(arr)]
}


// 对象数组按 key 去重
function uniqueByKey(arr, key) {
  const set = new Set()
  return arr.filter(item => {
    if (set.has(item[key])) return false
    set.add(item[key])
    return true
  })
}

// 对象数组按内容去重（JSON）
function uniqueByJSON(arr) {
  const set = new Set()
  return arr.filter(item => {
    const key = JSON.stringify(item)
    if (set.has(key)) return false
    set.add(key)
    return true
  })
}
```
